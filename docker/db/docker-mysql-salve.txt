docker run --name mysql-master \
-v /data/docker/mysql/conf/:/etc/mysql/mysql.conf.d/:rw \
-v /data/docker/mysql/logs:/var/log/mysql/:rw \
-v /data/docker/mysql/data/:/var/lib/mysql/:rw \
-e MYSQL_ROOT_PASSWORD=123456 \
-d mysql:5.7

docker exec -it mysql /bin/bash
GRANT ALL PRIVILEGES ON . TO 'root'@'%' WITH GRANT OPTION;
